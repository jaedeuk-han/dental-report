const SHEET_ID = '1bUtNE1cqfW9JC7JIKmzUwI9A1MxCTG09n4gUJFRTSv8'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); const action = data.action; if (action === 'save') { const sheet = ss.getSheetByName('data') || ss.insertSheet('data'); sheet.clearContents(); sheet.getRange(1, 1).setValue(JSON.stringify(data.payload)); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (action === 'load') { const sheet = ss.getSheetByName('data'); if (!sheet || sheet.getLastRow() === 0) return ContentService.createTextOutput(JSON.stringify({ok:true,payload:null})) .setMimeType(ContentService.MimeType.JSON); const val = sheet.getRange(1,1).getValue(); return ContentService.createTextOutput(JSON.stringify({ok:true,payload:JSON.parse(val)})) .setMimeType(ContentService.MimeType.JSON); } if (action === 'savelog') { const sheet = ss.getSheetByName(data.sheet) || ss.insertSheet(data.sheet); sheet.clearContents(); sheet.getRange(1, 1).setValue(JSON.stringify(data.payload)); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (action === 'loadlog') { const sheet = ss.getSheetByName(data.sheet); if (!sheet || sheet.getLastRow() === 0) return ContentService.createTextOutput(JSON.stringify({ok:true,payload:null})) .setMimeType(ContentService.MimeType.JSON); const val = sheet.getRange(1,1).getValue(); return ContentService.createTextOutput(JSON.stringify({ok:true,payload:JSON.parse(val)})) .setMimeType(ContentService.MimeType.JSON); } } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput('OK').setMimeType(ContentService.MimeType.TEXT); }
